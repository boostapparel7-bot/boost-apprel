Ikuti 3 langkah berikut untuk menghubungkan sistem ini dengan Google Sheets dan notifikasi WhatsApp otomatis.

1 Buat Google Sheet

Buat spreadsheet baru di Google Sheets. Beri nama sheet: Orders. Buat kolom header berikut di baris pertama:

ID | Timestamp | Nama | WhatsApp | Qty | Jenis | Harga | Deadline | Status | Catatan | Updated

Copy ID Spreadsheet dari URL: https://docs.google.com/spreadsheets/d/[SPREADSHEET_ID]/edit

2 Pasang Google Apps Script

Di Google Sheet, buka Extensions → Apps Script, hapus kode default, paste kode berikut:

const SHEET_ID = 'GANTI_DENGAN_SPREADSHEET_ID'; const WA_TOKEN = 'GANTI_DENGAN_FONNTE_TOKEN'; const SHEET_NAME = 'Orders'; function doGet(e) { try { const action = e.parameter.action; const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); if (action === 'getOrders') { const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const orders = rows.slice(1).map(r => { let obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); return ok({orders}); } return ok({error: 'Unknown action'}); } catch(err) { return ContentService.createTextOutput( JSON.stringify({error: err.message}) ).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); if (action === 'addOrder') { const newId = 'JR-' + String(sheet.getLastRow()).padStart(3,'0'); sheet.appendRow([ newId, new Date().toLocaleString('id-ID'), data.nama, data.wa, data.qty, data.jenis, data.harga, data.deadline, 'DP', data.catatan, new Date().toLocaleString('id-ID') ]); sendWA(data.wa, `✅ *ORDER DITERIMA*\n\nHalo ${data.nama}!\nOrder jersey ${data.jenis} Anda sudah tercatat.\n\n📋 *ID Order:* ${newId}\n👕 *Qty:* ${data.qty} pcs\n📅 *Deadline:* ${data.deadline}\n💰 *Total:* Rp ${data.harga}\n\n*Status saat ini: DP*\nKami akan update setiap ada perkembangan 🙏` ); return ok({id: newId}); } if (action === 'updateStatus') { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sheet.getRange(i+1, 9).setValue(data.status); sheet.getRange(i+1, 11).setValue(new Date().toLocaleString('id-ID')); const wa = rows[i][3]; const nama = rows[i][2]; const msg = getStatusMsg(nama, data.id, data.status); sendWA(wa, msg); return ok({updated: true}); } } return ok({updated: false, msg: 'ID tidak ditemukan'}); } if (action === 'deleteOrder') { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sheet.deleteRow(i+1); return ok({deleted: true}); } } return ok({deleted: false, msg: 'ID tidak ditemukan'}); } if (action === 'getOrders') { const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const orders = rows.slice(1).map(r => { let obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); return ok({orders}); } } catch(err) { return ContentService.createTextOutput( JSON.stringify({error: err.message}) ).setMimeType(ContentService.MimeType.JSON); } } function sendWA(nomor, pesan) { const url = 'https://api.fonnte.com/send'; UrlFetchApp.fetch(url, { method: 'POST', headers: {'Authorization': WA_TOKEN}, payload: {target: nomor, message: pesan} }); } function getStatusMsg(nama, id, status) { const emoji = { 'DP':'💰','Design':'🎨','Cetak':'🖨️', 'Jahit':'🧵','Packing':'📦','Selesai':'✅' }; const info = { 'DP': 'Menunggu konfirmasi DP.', 'Design': 'Tim desainer sedang mengerjakan layout jersey Anda.', 'Cetak': 'Desain sudah disetujui, masuk antrian cetak.', 'Jahit': 'Bahan sudah dicetak, sedang proses jahit.', 'Packing': 'Jersey selesai dijahit, sedang dikemas.', 'Selesai': 'Jersey siap diambil / dikirim! Terima kasih 🙏' }; return `${emoji[status]||'📋'} *UPDATE ORDER ${id}*\n\nHalo ${nama}!\nStatus order Anda sudah diperbarui.\n\n🔄 *Status: ${status.toUpperCase()}*\n${info[status]||''}\n\nInfo lebih lanjut hubungi kami ya! 😊`; } function ok(data) { return ContentService.createTextOutput( JSON.stringify({success:true, ...data}) ).setMimeType(ContentService.MimeType.JSON); }

Setelah paste kode: Deploy → New deployment → Web app → Execute as: Me → Access: Anyone → Deploy. Copy URL yang muncul.

3 Daftarkan Fonnte (WA Gateway)

Daftar di fonnte.com, connect WhatsApp Anda, lalu copy token API. Masukkan ke variabel WA_TOKEN di Apps Script di atas.
